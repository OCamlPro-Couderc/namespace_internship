\begin{frame}[fragile]
\frametitle{Before 4.02: developpers' trick \#2}

\emph{Packs}

\bigskip

\only<2-4>{
\begin{itemize}
\item<3-4> LibA (Misc, Map, ...)
\item<4-4> $\rightarrow$ LibA = a \alert{unique module} LibA
  
  with \alert{submodules}: (Misc, Map, ...)
\end{itemize}

\bigskip
}

\only<5->{
Pros:
\begin{itemize}
\item<5->\textbf{Developper}: Simply changes the compilation toolchain;
\item<6->\textbf{User}: 
  \begin{itemize}
  \item Use path to distinguish modules
  \item Use the module as any other module
  \end{itemize}
\end{itemize}
\medskip
}

\only<7->{
Cons:
\begin{itemize}
\item<7-> Too much recompilations
\item<8-> Dependencies
\item<9-> Binaries size
\end{itemize}
}

\end{frame}

\begin{frame}
Our solution: \emph{namespaces}.

\bigskip

Principle:
\begin{itemize}[<+->]
\item A module belongs to a namespace
\item Modules in namespaces are imported explicitely
\end{itemize}

\bigskip

\only<3>{
  To compare with Java or Scala's packages.
}
\end{frame}

\only<1>{
How to use those modules in my program?

\defverbatim[colored]{\simpleimport}{
\begin{OCaml}
with LibA.(Misc, Map)

open Misc (* from LibA *)
let empty = Map.empty
...
let _ = ... 
  Misc.pprint 42 (* function only in my own Misc *)
\end{OCaml}}

\only<2-3>{\simpleimport \bigskip}

\only<3>{

  signature: value pprint does not exists!''}}

\defverbatim[colored]{\aliasedimport}{
\begin{OCaml}
with LibA.(Misc as Misc2, Map)

open Misc2
let empty = Map.empty
...
let _ = ... 
  Misc.pprint 42
\end{OCaml}}

\only<4->{
  We need Misc and ``Misc from LibA'' at the same time:
  
  \bigskip

  \aliasedimport
}

\end{frame}

\begin{frame}[fragile]
\frametitle{Extending Pervasives to namespaces}

\only<1-2>{
Pervasives: automatically opened.
}

\defverbatim[colored]{\pervasives}{
\begin{OCaml}
with Stdlib.List
...

\end{OCaml}}

\only<2>{
\pervasives

\bigskip

$\rightarrow$ Pervasives modules automatically opened when using their namespace.
}

\defverbatim[colored]{\palias}{
\begin{OCaml}
in namespace Stdlib.(Pervasives as P)
\end{OCaml}
}

in namespace Stdlib.(Pervasives as _)
\end{OCaml}
}

\only<3-4>{
Preventing auto-opens: 
\bigskip
\begin{itemize}
\item<3-> By renaming:
\palias
\item<4-> And shadowing:

\begin{frame}
\frametitle{ocamldep and namespaces}

Namespace declaration and imports $\equiv$ header

\bigskip

Dependencies computed easier: each import \textbf{obviously is} a file.

\bigskip

With a large adoption and use of namespace: 

$\rightarrow$ ocamldep only needs to read the header.

\end{frame}

 $\rightarrow$ Description of the compilation environnement
